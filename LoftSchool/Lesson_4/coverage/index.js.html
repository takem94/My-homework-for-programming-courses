/**
 * Функция должна перебрать все дочерние элементы элемента where
 * и вернуть массив, состоящий из тех дочерних элементов
 * следующим соседом которых является элемент с тегом P
 * Рекурсия - по желанию
 *
 * @param {Element} where - где искать
 * @return {Array<Element>}
 *
 * @example
 * для html '<div></div><p></p><a></a><span></span><p></p>'
 * функция должна вернуть: [div, span]
 * т.к. следующим соседом этих элементов является элемент с тегом P
 */
function findAllPSiblings(where) {
    let result = [];
 
    for (let i = 0; i < where.children.length - 1; ++i) {
        let nextElem = where.children[i].nextElementSibling;
 
        if (nextElem.tagName === 'P') {
            result.push(where.children[i]);
        }
    }
 
    return result;
}

/**
 * *** Со звездочкой ***
 * Необходимо собрать статистику по всем узлам внутри элемента root и вернуть ее в виде объекта
 * Статистика должна содержать:
 * - количество текстовых узлов
 * - количество элементов каждого класса
 * - количество элементов каждого тега
 * Для работы с классами рекомендуется использовать свойство classList
 * Постарайтесь не создавать глобальных переменных
 *
 * @param {Element} root - где собирать статистику
 * @return {{tags: Object<string, number>, classes: Object<string, number>, texts: number}}
 *
 * @example
 * для html <div class="some-class-1"><b>привет!</b> <b class="some-class-1 some-class-2">loftschool</b></div>
 * должен быть возвращен такой объект:
 * {
 *   tags: { DIV: 1, B: 2},
 *   classes: { "some-class-1": 2, "some-class-2": 1 },
 *   texts: 3
 * }
 */
function collectDOMStat(root) {
    let result = {
        texts: 0,
        tags: {},
        classes: {}
    };
 
    let compare = function (what, result) {
        if (result !== undefined) {
            return (result + what);
        }
 
        return what;
    };
 
    for (let i = 0; i < root.childNodes.length; ++i) {
        let elem = root.childNodes[i];
 
        if (elem.nodeType === 3) {
            ++result.texts;
        } else {
 
            result.tags[elem.tagName] = compare(1, result.tags[elem.tagName]);
 
            for (let j = 0; j < elem.classList.length; ++j) {
                result.classes[elem.classList[j]] = compare(1, result.classes[elem.classList[j]]);
            }
 
            let tempResult = collectDOMStat(elem);
 
            for (let key in tempResult.tags) {
                result.tags[key] = compare(tempResult.tags[key], result.tags[key]);
            }
 
            for (let key in tempResult.classes) {
                result.classes[key] = compare(tempResult.classes[key], result.classes[key]);
            }
 
            result.texts += tempResult.texts;
        }
    }
 
    return result;
}
